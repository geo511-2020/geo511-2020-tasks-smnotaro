# Preparing Dot Plot

``` r
groups <- as.factor(cropped$variable)
dotprep <- foreach(i=1:4, .combine = "rbind", .packages = c("tidyverse", "sf")) %dopar% {
  races <- levels(groups)[i]
cropped %>%
    filter(variable == races) %>%
    st_sample(size = .$value) %>%
    st_as_sf() %>%
    mutate(variable = races)
}
```
